const ComponentLibrary = (props: Props) => {
 
    const nestedForms = lib.map(item => {
        const inputs = item.inputs(item.state);
 
        const label: NavigationFormInputConfig['label'] = !item.description
            ? item.name
            : ({ expand }) => {
                return (
                    <DescriptiveNavigationLabel 
                        testID={fauxTestId}
                        expand={expand} 
                        name={item.name} 
                        description={item.description} />
                )
            }
 
            
 
        const navigationInputConfig: NavigationFormInputConfig = {
            testID: fauxTestId,
            name: item.name,
            label: label,
            icon: item.icon,
            screen: ({ back }) => {
                return <Form testID={fauxTestId} headerLabel={item.name} inputs={inputs} submit={{ label: 'Done', handler: async () => back() }}/>
            }
        }
 
        return navigationInputConfig
    });
 
    return (
        <View style={{ flex: 1, backgroundColor: '#fff' }}>
            <Form testID={fauxTestId} inputs={nestedForms} headerLabel={'Demos'}/>
        </View>
    )
    
}
